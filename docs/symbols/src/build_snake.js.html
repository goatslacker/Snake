 26   /**
 27     * Inserts a foreign object into a template.
 28     *
 29     * @param {string} str The string to interpolate
 30     * @param {Object} obj The foreign Object to interpolate into the string
 31     * @returns {string} The string interpolated with the object's values
 32     */
 33   interpolate: function (str, obj) {
 34     for (var prop in obj) {
 35       if (obj.hasOwnProperty(prop)) {
 36         str = str.replace(new RegExp('#{' + prop + '}', 'g'), typeof obj[prop][1] === 'f' ? obj[prop]() : obj[prop]);
 37       }
 38     }
 39 
 40     return str;
 41   },

 43   /**
 44     * Dynamically builds the Models
 45     *
 46     * @param {Object} schema The schema in JSON format
 47     * @param {Function} onComplete The callback function to execute once the schema finishes building
 48     */
 49   loadFromJSON: function (schema, onComplete) {
 50     var table = null,
 51         column = null,
 52         def_column = null,
 53         fk = null,
 54         model = null;
 55 
 56     for (table in schema) {
 57       if (schema.hasOwnProperty(table)) {
 58         model = schema[table];
 59 
 60         model.jsName = table;
 61         model.columns.id = { type: "INTEGER" };
 62         model.columns.created_at = { type: "TIME" };
 63 
 64         model.map = [];
 65         for (column in schema[table].columns) {
 66           if (schema[table].columns.hasOwnProperty(column)) {
 67             def_column = schema[table].columns[column];
 68 
 69             if ("foreign" in def_column) {
 70               if (!model.foreign) {
 71                 model.foreign = {};
 72               }
 73 
 74               fk = def_column.foreign.split(".");
 75               model.foreign[fk[0]] = [column, fk[1]];
 76             }
 77 
 78             model.map.push(column);
 79           }
 80         }
 81 
 82         // TODO create relationships for the base objects
 83         // TODO create doSelectJoins for the relationships
 84 
 85         Snake.venom[schema[table].tableName] = new Snake.venomousObject(model);
 86         Snake.global[table] = new Snake.base(model);
 87       }
 88     }
 89 
 90     if (onComplete) {
 91       onComplete();
 92     }
 93

179   /** @lends Model.prototype */

181     /**
182       * Saves a record to the database
183       *
184       * @param {Function} onSuccess The callback to execute if the transaction completes successfully
185       * @param {Function} onFailure The callback to execute if the transaction fails
186       * @param {boolean} outputSql If true the SQL is returned to the onSuccess callback as a string, otherwise the data is persisted to the database
187       */
188     save: function (onSuccess, onFailure, outputSql) {
189       var model = this,
190           values = [],
191           interpolate = Snake.interpolate,
192           q = [],
193           i = 0,
194           max = 0,
195           val = null,
196           sql = "";
197 
198       // update
199       if (this.id) {
200         for (i = 0, max = table.columns.length; i < max; i = i + 1) {
201           if (this[table.columns[i]] !== this.old[table.columns[i]]) {
202             val = this[table.columns[i]] || null;
203             values.push(val);
204 
205             q.push(table.columns[i] + " = ?");
206           }
207         }
208 
209         sql = interpolate("UPDATE #{table} SET #{conditions} WHERE id = #{id}", {
210           table: table.tableName,
211           conditions: q,
212           id: this.id
213         });
214 
215       // insert
216       } else {
217 
218         for (i = 0, max = table.map.length; i < max; i = i + 1) {
219           val = this[table.map[i]] || null;
220   
221           if (table.map[i] === 'created_at' && val === null) {

223           }
224 
225           values.push(val);
226           q.push("?");
227         }
228 
229         sql = interpolate("INSERT INTO '#{table}' (#{columns}) VALUES (#{q})", {
230           table: table.tableName,
231           columns: table.map,
232           q: q
233         });
234       }
235 
236 
237       if (outputSql === true) {
238         if (onSuccess) {
239           onSuccess(sql, values);
240         }
241 
242       } else {
243         Snake.query(sql, values, function (transaction, results) {
244           // set an ID
245           model.id = results.insertId;
246 
247           if (onSuccess) {
248             onSuccess(model);
249           }

251       }

274 /**
275   * Performs a query on the Web Database
276   *
277   * @function
278   * @param {string} query A prepared statement
279   * @param {Array} params The parameters to insert into the prepared statements
280   * @param {Function} onSuccess The function to callback if the transaction is successfully executed
281   * @param {Function} onFailure The function to callback if the transaction fails
282   */
283 Snake.query = (function () {

286 /**
287   * @private
288   */
289   var Database = null,

292 /**
293   * @private
294   */
295   function connect(onSuccess, onFailure) {
296     var self = Snake,
297         db = self.config.database;
298 
299     // defaults
300     onSuccess = onSuccess || function () {};
301     onFailure = onFailure || function () {};
302 
303     // HTML5 openDatabase
304     Database = openDatabase(db.name, db.version, db.displayName, db.size);
305 
306     // callbacks
307     if (!Database) {
308       onFailure("Could not open database");
309     } else {
310       onSuccess();
311     }
312   }

314   /**
315     * @private
316     */
317   Query = function (query, params, onSuccess, onFailure) {
318     var self = Snake;
319 
320     // defaults
321     params = params || null;
322 
323     onSuccess = onSuccess || function (transaction, results) {
324       self.log(transaction);
325       self.log(results);
326     };
327     onFailure = onFailure || function (transaction, error) {
328       self.log(transaction);
329       self.log(error);
330     };
331 
332     if (!Database) {
333       self.log("Connecting to the database");
334       connect(function () {
335         Snake.query(query, params, onSuccess, onFailure);
336       });
337     } else {
338     
339       // HTML5 database perform query
340       Database.transaction(function (transaction) {
341         var preparedQuery = null,
342             i = 0,
343             max = 0;
344 
345         // convert to single array
346         if (!self.is_array(query)) {
347           query = [query];
348         }
349 
350         for (i, max = query.length; i < max; i = i + 1) {
351 
352           // append semicolon to query
353           preparedQuery = query[i] + ";";
354 
355           // debugging
356           if (self.debug) {
357             self.log(preparedQuery);
358             if (params) {
359               self.log(params);
360             }
361           }
362 
363           // perform query
364           transaction.executeSql(preparedQuery, params, onSuccess, onFailure);
365         }
366 
367       });
368     }
369   };

371   return Query;

374 /**
375   * Creates the `Query Building` object which is used to extract data from the database
376   *
377   * @constructor
378   * @param {Object} schema The object's schema
379   * @returns {Object}
380   */
381 Snake.venomousObject = function (schema) {
382   /**
383     * @private
384     */
385   var Selectors = {},
386       Collection = {},
387       queryBuilder = null,
388       addWhere = null,
389       resetObj = null;

391   /**
392     * @private
393     * Resets the Query after it returns a result
394     */
395   resetObj = function () {
396     Collection.sql = {
397       select: [],
398       from: schema.tableName,
399       joins: [],
400       where: {
401         criterion: [],
402         params: []
403       },
404       orderBy: [],
405       limit: false
406     };
407   };

409   /**
410     * @private
411     * Adds a WHERE statement to the query
412     */
413   addWhere = function () {
414     var field = arguments[0],
415         value = arguments[1],
416         selector = arguments[2] || Selectors.EQUAL,
417         q = [],
418         i = 0,
419         max = 0;
420 
421     if (field in schema.columns) {
422       field = schema.tableName + "." + field;
423     }
424 
425     switch (selector) {
426     case Selectors.ISNULL:
427     case Selectors.ISNOTNULL:
428       Collection.sql.where.criterion.push(field + " " + selector);
429       break;
430 
431     case Selectors.IN:
432     case Selectors.NOTIN:
433       for (i = 0, max = value.length; i < max; i = i + 1) {
434         q.push("?");
435       }
436 
437       Collection.sql.where.criterion.push(field + " " + selector + " (" + q.join(", ") + ")");
438       break;
439 
440     default:
441       Collection.sql.where.criterion.push(field + " " + selector + " ?");
442     }
443 
444     if (value) {
445       if (Snake.is_array(value)) {
446         Collection.sql.where.params = Collection.sql.where.params.concat(value);
447       } else {
448         Collection.sql.where.params.push(value);
449       }
450     }
451   };
452 
453   /**
454     * @private
455     * Builds the query and passes it onto Snake.query for processing
456     */
457   queryBuilder = function (persist, sql, query, onSuccess, onFailure) {
458     var params = null,
459         interpolate = Snake.interpolate;
460 
461     query = query || {};
462 
463     // FROM
464     query.from = schema.tableName;
465 
466     if (Collection.sql.joins.length > 0) {
467       sql = sql + " " + Collection.sql.joins.join(" ");
468     }
469 
470     // WHERE
471     if (Collection.sql.where.criterion.length > 0) {
472       sql = sql + " WHERE #{where}";
473       // build the where...
474       query.where = Collection.sql.where.criterion.join(" AND ");
475 
476       params = Collection.sql.where.params;
477     }
478 
479     // ORDER BY
480     if (Collection.sql.orderBy.length > 0) {
481       sql = sql + " ORDER BY #{orderBy}";
482       query.orderBy = Collection.sql.orderBy;
483     }
484 
485     // LIMIT && OFFSET
486     if (Collection.sql.limit) {
487       if (Collection.sql.offset) {
488         sql = sql + " LIMIT #{offset}, #{limit}";
489         query.offset = Collection.sql.offset;
490       } else {
491         sql = sql + " LIMIT #{limit}";
492       }
493 
494       query.limit = Collection.sql.limit;
495     }
496 
497     // We run the query
498     if (persist) {
499       Snake.query(interpolate(sql, query), params, onSuccess, onFailure);
500 
501     // use the callback to return the query
502     } else {
503       if (onSuccess) {
504         onSuccess(interpolate(sql, query), params);
505       }
506     }
507 
508     resetObj();
509   };

511   /**
512     * @private
513     * @constant
514     */
515   Selectors = {
516     EQUAL: "=", 
517     NOT_EQUAL: "<>",
518     GREATER_THAN: ">", 
519     LESS_THAN: "<", 
520     GREATER_EQUAL: ">=", 
521     LESS_EQUAL: "<=",
522     ISNULL: "IS NULL",
523     ISNOTNULL: "IS NOT NULL",
524     LIKE: "LIKE",
525     NOTLIKE: "NOT LIKE",
526     "IN": "IN",
527     NOTIN: "NOT IN",
528     LEFT_JOIN: "LEFT JOIN"
529   };
530 
531   /**
532     * @public
533     * @constructor
534     * @this {Collection} The collection object - in order to chain calls
535     */
536   Collection = {
537 
538     /**
539       * Adds select columns to the query
540       *
541       * @param {string} args The field names to select
542       * @example
543       * vql.fruits.select(id, tree_id, name, description).doSelect(callback);
544       * @returns {Object} this 
545       */
546     select: function () {
547       for (var i = 0, max = arguments.length; i < max; i = i + 1) {
548         if (arguments[i] in schema.columns) {
549           this.sql.select.push(schema.tableName + "." + arguments[i]);
550         }
551       }
552 
553       return this;
554     },
555 
556     /**
557       * Filters the results by the criteria specified
558       *
559       * @param {string} args The field names to select
560       * @example
561       * vql.fruits.find({ name: "mango" }).doCount(callback);
562       * vql.fruits.find("name", "mango").doCount(callback);
563       * @returns {Object} this 
564       */
565     find: function () {
566       var field = null,
567           value = null,
568           selector = null,
569           tmp = null;
570 
571       // if we're passing each argument
572       if (arguments.length > 1) {
573         // first argument is the field
574         field = arguments[0];
575         // second argument should be the value
576         value = arguments[1];
577 
578         // unless the value is actually a selector
579         if (value in Selectors) {
580           selector = Selectors[value];
581 
582         // otherwise the third argument is the selector
583         } else {
584           selector = Selectors[arguments[2]] || Selectors.EQUAL;
585         }
586 
587         addWhere(field, value, selector);
588 
589       // we're not passing each argument
590       } else {
591 
592         // Pull by ID
593         if (typeof(arguments[0]) === "number") {
594 
595           addWhere("id", arguments[0]);
596 
597         // It's an object
598         } else {
599 
600           // loop through each field
601           for (field in arguments[0]) {
602 
603             if (arguments[0].hasOwnProperty(field)) {
604               // the value is the property of the field
605               value = arguments[0][field];
606 
607               switch (Object.prototype.toString.call(value)) {
608               // if the value is an Array then we perform an IN query
609               case "[object Array]":
610                 selector = Selectors.IN;
611                 addWhere(field, value, selector);
612                 break;
613 
614               // if the value is a Regular Expression then we perform a LIKE query
615               case "[object RegExp]": 
616                 // TODO - NOT LIKE
617                 selector = Selectors.LIKE;
618                 tmp = value.toString();
619                 value = tmp;
620                 tmp = value.replace(/\W/g, "");
621 
622                 if (value.substr(1, 1) === '^') {
623                   value = tmp + '%';
624                 } else if (value.substr(-2, 1) === '$') {
625                   value = '%' + tmp;
626                 } else {
627                   value = '%' + tmp + '%';
628                 }
629 
630                 addWhere(field, value, selector);
631                 break;
632 
633               // if the value is an Object then we need to loop through all the items in the object and set them for the current field
634               case "[object Object]":
635                 for (tmp in value) {
636                   if (value.hasOwnProperty(tmp)) {
637                     selector = Selectors[tmp] || Selectors.EQUAL;
638 
639                     addWhere(field, value[tmp], selector);
640                   }
641                 }
642                 break;
643 
644               // by default the selector is =
645               default:
646                 selector = Selectors.EQUAL;
647                 addWhere(field, value, selector);
648               }
649             }
650 
651           } // loop
652 
653         } // typeof num
654 
655       } // endif
656 
657       return this;
658     },
659 
660     /**
661       * Orders the result set in ascending or descending order by a column
662       *
663       * @param {Object} obj The fields to order by along with their order
664       * @example
665       * vql.fruits.orderBy({ id: 'desc' }).doSelect(callback);
666       * @returns {Object} this 
667       */
668     orderBy: function (obj) {
669       var column = null,
670           sortOrder = "";
671       for (column in obj) {
672         if (obj.hasOwnProperty(column)) {
673           sortOrder = obj[column].toUpperCase();
674           if (column in schema.columns) {
675             column = schema.tableName + "." + column;
676           }
677           this.sql.orderBy.push(column + " " + sortOrder);
678         }
679       }
680 
681       return this;
682     },

684     /**
685       * Joins two tables together using the table's primary and foreign keys
686       * TODO params
687       * @returns {Object} this 
688       */
689     join: function (table, on, join_method) {
690       var interpolate = Snake.interpolate;
691 
692       join_method = Selectors[join_method] || Selectors.LEFT_JOIN;
693 
694       // find relationship and join the tables
695       if (!on) {
696         // this.join(vql.Deck);
697         if ("foreign" in schema && table in schema.foreign) {
698           this.sql.joins.push(interpolate("#{join_method} #{foreign_table} ON #{table}.#{primary_key} = #{foreign_table}.#{foreign_key}", {
699             join_method: join_method,
700             foreign_table: table,
701             table: schema.tableName,
702             primary_key: schema.foreign[table][0],
703             foreign_key: schema.foreign[table][1]
704           }));
705         }
706       // join it on the parameters provided
707       } else {
708         this.sql.joins.push(interpolate("#{join_method} #{foreign_table} ON #{table}.#{primary_key} = #{foreign_table}.#{foreign_key}", {
709           join_method: join_method,
710           foreign_table: table,
711           table: schema.tableName,
712           primary_key: on[0],
713           foreign_key: on[1]
714         }));
715       }
716 
717       return this;
718     },

744       *

746       * @param {Function} onSuccess The function to callback once the operation completes successfully

754     /**
755       * Retrieves one record from the database from the specified criteria 
756       *
757       * @param {Function} onSuccess The function to callback once the operation completes successfully
758       * @param {Function} onFailure The function to callback if the operation fails
759       * @param {boolean} outputSql If true the SQL is returned to the onSuccess callback as a string, otherwise we attempt to retrieve the data from the database
760       */
761     doSelectOne: function (onSuccess, onFailure, outputSql) {
762       var callback = null;
763 
764       if (outputSql === true) {
765         callback = onSuccess;
766       } else {
767         /** @private */
768         callback = function (rows) {
769           if (onSuccess) {
770             if (rows.length > 0) {
771               onSuccess(rows[0]);
772             } else {
773               onSuccess(null);
774             }
775           }
776         };
777       }
778 
779       this.limit(1).doSelect(callback, onFailure, outputSql);
780     },

782     /**
783       * Returns the number of records for a given criteria
784       *
785       * @param {Function} onSuccess The function to callback once the operation completes successfully
786       * @param {Function} onFailure The function to callback if the operation fails
787       * @param {boolean} useDistinct If true the COUNT is performed as distinct
788       * @param {boolean} outputSql If true the SQL is returned to the onSuccess callback as a string, otherwise we attempt to retrieve the data from the database
789       */
790     doCount: function (onSuccess, onFailure, useDistinct, outputSql) {
791       useDistinct = (useDistinct && this.sql.select.length > 0) ? "DISTINCT " : "";
792       var sql = "SELECT COUNT(" + useDistinct + "#{select}) AS count FROM #{from}",
793           callback = null,
794           query = {};
795 
796       if (this.sql.select.length === 0) {
797         query.select = "*";
798       } else {
799         query.select = this.sql.select;
800       }
801 
802       if (outputSql === true) {
803         callback = onSuccess;
804       } else {
805         /** @private */
806         callback = function (transaction, results) {
807           var obj = results.rows.item(0);
808 
809           if (onSuccess) {
810             onSuccess(obj.count);
811           }
812         };
813       }
814 
815       queryBuilder(!outputSql, sql, query, callback, onFailure);
816     },
817 
818     /**
819       * Deletes an object from the database
820       *
821       * @param {Function} onSuccess The function to callback once the operation completes successfully
822       * @param {Function} onFailure The function to callback if the operation fails
823       * @param {boolean} outputSql If true the SQL is returned to the onSuccess callback as a string, otherwise we attempt to retrieve the data from the database
824       */
